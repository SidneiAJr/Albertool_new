# 🗄️ SQL Starter – Gerador de Scripts SQL

> Cansado de escrever `CREATE TABLE` na mão toda vez? Esse CLI resolve.

---

## 📌 Sobre o Projeto

**SQL Starter** é uma CLI simples que gera scripts SQL prontos para diferentes bancos de dados.  
Você escolhe o banco, informa o nome da tabela, e o script sai pronto com:

- Criação do banco e tabela
- Chave primária auto-increment
- Campos `created_at` e `updated_at` com trigger
- Dados de teste iniciais

## 📦 Como Usar

1. Crie uma pasta e dentro dela um arquivo `setup.sh`
2. Cole o script do repositório
3. Execute com gitbash no windows ou no terminal no linux

## Script:
```bash
#!/bin/bash
# sql-starter.sh
# Gerador de scripts SQL para diferentes bancos

echo "======================================"
echo "SQL Starter - Gerador de Scripts SQL"
echo "CLI MADE IN BRASIL"
echo "======================================"

read -p "Nome do banco: " DB_NAME
read -p "Tabela principal: " TABLE_NAME

# Diretórios
BASE_DIR="$DB_NAME"
BACKEND_DIR="$BASE_DIR/sql"

mkdir -p "$BACKEND_DIR"

echo "📁 Projeto criado em: $BASE_DIR"

# ================================
# MySQL
# ================================
criar_sql_mysql(){
    cat <<EOF > "$BACKEND_DIR/${DB_NAME}_mysql.sql"
-- Database: $DB_NAME
CREATE DATABASE IF NOT EXISTS $DB_NAME;
USE $DB_NAME;

-- Tabela: $TABLE_NAME
CREATE TABLE IF NOT EXISTS $TABLE_NAME (
    id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    nome VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Inserir dados de teste
INSERT INTO $TABLE_NAME (nome) VALUES ('Exemplo 1'), ('Exemplo 2');
EOF
}

# ================================
# SQL Server
# ================================
criar_sql_server(){
    cat <<EOF > "$BACKEND_DIR/${DB_NAME}_sqlserver.sql"
-- Database: $DB_NAME
CREATE DATABASE $DB_NAME;
GO

USE $DB_NAME;
GO

-- Tabela: $TABLE_NAME
CREATE TABLE $TABLE_NAME (
    id INT IDENTITY(1,1) PRIMARY KEY,
    nome NVARCHAR(100) NOT NULL,
    created_at DATETIME DEFAULT GETDATE(),
    updated_at DATETIME DEFAULT GETDATE()
);
GO

-- Trigger para updated_at
CREATE TRIGGER trg_${TABLE_NAME}_update
ON $TABLE_NAME
AFTER UPDATE
AS
BEGIN
    UPDATE $TABLE_NAME
    SET updated_at = GETDATE()
    WHERE id IN (SELECT DISTINCT id FROM inserted);
END;
GO

-- Inserir dados de teste
INSERT INTO $TABLE_NAME (nome) VALUES ('Exemplo 1'), ('Exemplo 2');
GO
EOF
}

# ================================
# Oracle
# ================================
criar_sql_oracle(){
    cat <<EOF > "$BACKEND_DIR/${DB_NAME}_oracle.sql"
-- Database: $DB_NAME (criar manualmente no Oracle)
-- Tabela: $TABLE_NAME
CREATE TABLE $TABLE_NAME (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome VARCHAR2(100) NOT NULL,
    created_at DATE DEFAULT SYSDATE,
    updated_at DATE DEFAULT SYSDATE
);

-- Trigger para updated_at
CREATE OR REPLACE TRIGGER trg_${TABLE_NAME}_update
BEFORE UPDATE ON $TABLE_NAME
FOR EACH ROW
BEGIN
    :NEW.updated_at := SYSDATE;
END;
/

-- Inserir dados de teste
INSERT INTO $TABLE_NAME (nome) VALUES ('Exemplo 1');
INSERT INTO $TABLE_NAME (nome) VALUES ('Exemplo 2');
COMMIT;
EOF
}

# ================================
# PostgreSQL
# ================================
criar_sql_postgres(){
    cat <<EOF > "$BACKEND_DIR/${DB_NAME}_postgres.sql"
-- Database: $DB_NAME
CREATE DATABASE $DB_NAME;
\c $DB_NAME;

-- Tabela: $TABLE_NAME
CREATE TABLE $TABLE_NAME (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Função para updated_at
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS \$\$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
\$\$ LANGUAGE plpgsql;

CREATE TRIGGER trg_${TABLE_NAME}_update
BEFORE UPDATE ON $TABLE_NAME
FOR EACH ROW
EXECUTE FUNCTION update_updated_at_column();

-- Inserir dados de teste
INSERT INTO $TABLE_NAME (nome) VALUES ('Exemplo 1'), ('Exemplo 2');
EOF
}

# ================================
# SQLite
# ================================
criar_sql_sqlite(){
    cat <<EOF > "$BACKEND_DIR/${DB_NAME}_sqlite.sql"
-- Database: $DB_NAME (SQLite usa arquivo)
-- Tabela: $TABLE_NAME
CREATE TABLE IF NOT EXISTS $TABLE_NAME (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    nome TEXT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Trigger para updated_at
CREATE TRIGGER IF NOT EXISTS trg_${TABLE_NAME}_update
AFTER UPDATE ON $TABLE_NAME
BEGIN
    UPDATE $TABLE_NAME SET updated_at = CURRENT_TIMESTAMP WHERE id = NEW.id;
END;

-- Inserir dados de teste
INSERT INTO $TABLE_NAME (nome) VALUES ('Exemplo 1'), ('Exemplo 2');
EOF
}

# ================================
# Azure SQL
# ================================
criar_sql_azure(){
    criar_sql_server  # Azure SQL usa sintaxe muito parecida com SQL Server
    mv "$BACKEND_DIR/${DB_NAME}_sqlserver.sql" "$BACKEND_DIR/${DB_NAME}_azure.sql"
}

# ================================
# MENU
# ================================
echo ""
echo "========= MENU ========="
echo "1 | MySQL"
echo "2 | SQL Server"
echo "3 | Oracle"
echo "4 | PostgreSQL"
echo "5 | SQLite"
echo "6 | Azure SQL"
echo "0 | Sair"
echo "========================"
read -p "Escolha o banco: " OPT

case $OPT in
  1) criar_sql_mysql ;;
  2) criar_sql_server ;;
  3) criar_sql_oracle ;;
  4) criar_sql_postgres ;;
  5) criar_sql_sqlite ;;
  6) criar_sql_azure ;;
  0) exit 0 ;;
  *) echo "Opção inválida!" ;;
esac

echo ""
echo "✅ Script SQL gerado em: $BACKEND_DIR/${DB_NAME}_*.sql"
echo "   cd $BASE_DIR/sql"
````
